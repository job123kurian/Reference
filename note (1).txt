## .github/workflows/deploy-sandbox.yml
```yaml
name: Deploy to Sandbox

on:
  push:
    branches: [develop]

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment: sandbox
    permissions:
      id-token: write
      contents: read

    steps:
      - uses: actions/checkout@v3

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v2
        with:
          role-to-assume: arn:aws:iam::435097195861:role/lz-glo-iam-rol-c_pa
          aws-region: eu-north-1

      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v2
        with:
          terraform_version: 1.6.0

      - name: Terraform Init
        working-directory: infrastructure/environments/sandbox
        run: terraform init

      - name: Terraform Plan
        working-directory: infrastructure/environments/sandbox
        run: |
          terraform plan \
            -var="db_user=${{ secrets.DB_USER }}" \
            -var="db_password=${{ secrets.DB_PASSWORD }}"

      - name: Terraform Apply
        working-directory: infrastructure/environments/sandbox
        run: |
          terraform apply -auto-approve \
            -var="db_user=${{ secrets.DB_USER }}" \
            -var="db_password=${{ secrets.DB_PASSWORD }}"

      - name: Package Code
        run: |
          cd src
          zip -r ../code.zip . \
            --exclude "*.pyc" \
            --exclude "__pycache__/*" \
            --exclude ".env"
          cd ..
          echo "✅ Code packaged"

      - name: Upload to S3
        run: |
          BUCKET="mpan-validation-codebuild-sandbox-435097195861"
          aws s3 cp code.zip s3://$BUCKET/code.zip
          aws s3 cp docker/tnsnames.ora s3://$BUCKET/tnsnames.ora
          zip source.zip docker/Dockerfile buildspec.yml
          aws s3 cp source.zip s3://$BUCKET/source.zip
          echo "✅ Files uploaded to S3"

      - name: Trigger CodeBuild
        run: |
          BUILD_ID=$(aws codebuild start-build \
            --project-name mpan-validation-build-sandbox \
            --region eu-north-1 \
            --query 'build.id' \
            --output text)

          echo "Build ID: $BUILD_ID"

          while true; do
            STATUS=$(aws codebuild batch-get-builds \
              --ids $BUILD_ID \
              --region eu-north-1 \
              --query 'builds[0].buildStatus' \
              --output text)
            echo "$(date): $STATUS"
            [ "$STATUS" = "SUCCEEDED" ] || \
            [ "$STATUS" = "FAILED" ]    || \
            [ "$STATUS" = "STOPPED" ]   && break
            sleep 30
          done

          if [ "$STATUS" != "SUCCEEDED" ]; then
            echo "❌ Build failed!"
            exit 1
          fi

          echo "✅ Sandbox deployment complete!"
```

    logger.info("="*80)
    
    # 1. Validate event has Records
    if "Records" not in event:
        logger.error("No Records in event - must be triggered by S3")
        raise ValueError("Event must contain Records - trigger via S3 upload")
    
    # 2. Extract S3 file info
    record = event["Records"][0]
    src_bucket = record["s3"]["bucket"]["name"]
    src_key = record["s3"]["object"]["key"]
    
    logger.info(f"Input bucket: {src_bucket}")
    logger.info(f"Input file:   {src_key}")
    
    # 3. Download file to /tmp
    filename = os.path.basename(src_key)
    local_input = f"/tmp/{filename}"
    
    logger.info(f"Downloading to {local_input}...")
    s3.download_file(src_bucket, src_key, local_input)
    logger.info("✅ File downloaded")
    
    # 4. Define output path
    output_filename = filename.replace(".xlsx", "_Updated.xlsx")
    local_output = f"/tmp/{output_filename}"
    
    # 5. Process file
    logger.info("Starting processing...")
    output_path = process_allocation_file(local_input, local_output)
    
    if not output_path:
        raise Exception("Processing failed - no output generated")
    
    logger.info(f"✅ Processing complete: {output_path}")
    
    # 6. Upload output to S3
    dest_bucket = os.environ["OUTPUT_BUCKET"]
    dest_key = f"results/{os.path.basename(output_path)}"
    
    logger.info(f"Uploading to s3://{dest_bucket}/{dest_key}...")
    s3.upload_file(output_path, dest_bucket, dest_key)
    logger.info("✅ Output uploaded")
    
    # 7. Cleanup /tmp
    try:
        os.remove(local_input)
        os.remove(output_path)
    except:
        pass
    
    logger.info("✅ MPAN Validation complete!")
    
    return {
        "status": "success",
        "input_file": src_key,
        "output_file": dest_key,
        "output_bucket": dest_bucket,
        "environment": os.environ.get("ENVIRONMENT", "unknown")
    }
