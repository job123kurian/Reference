## buildspec.yml
```yaml
version: 0.2

phases:
  pre_build:
    commands:
      - echo Logging in to ECR...
      - aws ecr get-login-password --region $AWS_REGION | docker login --username AWS --password-stdin $ECR_REPO_URL
      - aws ecr create-repository --repository-name mpan-validation --region $AWS_REGION 2>/dev/null || echo "Repo exists"
      - echo Downloading files from S3...
      - aws s3 cp s3://$SOURCE_BUCKET/code.zip ./code.zip
      - aws s3 cp s3://$SOURCE_BUCKET/tnsnames.ora ./tnsnames.ora

  build:
    commands:
      - echo Building Docker image...
      - docker build -t mpan-validation:latest -f docker/Dockerfile .
      - docker tag mpan-validation:latest $ECR_REPO_URL:latest
      - docker tag mpan-validation:latest $ECR_REPO_URL:$CODEBUILD_BUILD_NUMBER

  post_build:
    commands:
      - echo Pushing to ECR...
      - docker push $ECR_REPO_URL:latest
      - docker push $ECR_REPO_URL:$CODEBUILD_BUILD_NUMBER
      - echo Deploying to Lambda...
      - aws lambda update-function-code --function-name $LAMBDA_FUNCTION_NAME --image-uri $ECR_REPO_URL:latest --region $AWS_REGION
      - aws lambda wait function-updated --function-name $LAMBDA_FUNCTION_NAME --region $AWS_REGION
      - echo Done!
```

## docker/Dockerfile
```dockerfile
FROM public.ecr.aws/lambda/python:3.12

RUN dnf install -y unzip libaio wget && dnf clean all

RUN mkdir -p /opt/oracle && \
    wget -q https://download.oracle.com/otn_software/linux/instantclient/2115000/instantclient-basic-linux.x64-21.15.0.0.0dbru.zip \
    -O /tmp/oracle.zip && \
    unzip /tmp/oracle.zip -d /opt/oracle && \
    rm /tmp/oracle.zip

ENV LD_LIBRARY_PATH=/opt/oracle/instantclient_21_15:$LD_LIBRARY_PATH
ENV ORACLE_HOME=/opt/oracle/instantclient_21_15
ENV TNS_ADMIN=/opt/oracle/instantclient_21_15/network/admin

WORKDIR /var/task

COPY code.zip /tmp/code.zip
RUN unzip /tmp/code.zip -d /tmp/code_extracted && \
    cp -r /tmp/code_extracted/* /var/task/ && \
    rm -rf /tmp/code_extracted /tmp/code.zip && \
    ls -al /var/task/

COPY tnsnames.ora /opt/oracle/instantclient_21_15/network/admin/tnsnames.ora

RUN pip install --no-cache-dir -r requirements.txt

RUN echo 'import pandas; print("pandas OK:", pandas.__version__)' > /tmp/verify.py && \
    echo 'import oracledb; print("oracledb OK:", oracledb.__version__)' >> /tmp/verify.py && \
    echo 'import boto3; print("boto3 OK:", boto3.__version__)' >> /tmp/verify.py && \
    python /tmp/verify.py && \
    rm /tmp/verify.py

RUN echo 'import oracledb; oracledb.init_oracle_client(lib_dir="/opt/oracle/instantclient_21_15"); print("Oracle thick mode OK")' > /tmp/oracle_check.py && \
    python /tmp/oracle_check.py && \
    rm /tmp/oracle_check.py

RUN cd /var/task && python -c "import sys; sys.path.insert(0, '/var/task'); import lambda_function; print('lambda_function OK')"

CMD ["lambda_function.lambda_handler"]
```

## .github/workflows/deploy-sandbox.yml
```yaml
name: Deploy to Sandbox

on:
  push:
    branches: [develop]

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment: sandbox
    permissions:
      id-token: write
      contents: read

    steps:
      - uses: actions/checkout@v3

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v2
        with:
          role-to-assume: arn:aws:iam::435097195861:role/lz-glo-iam-rol-c_pa
          aws-region: eu-north-1

      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v2
        with:
          terraform_version: 1.6.0

      - name: Terraform Init
        working-directory: infrastructure/environments/sandbox
        run: terraform init

      - name: Terraform Apply
        working-directory: infrastructure/environments/sandbox
        run: |
          terraform apply -auto-approve \
            -var="db_user=${{ secrets.DB_USER }}" \
            -var="db_password=${{ secrets.DB_PASSWORD }}"

      - name: Package Code
        run: |
          cd src
          zip -r ../code.zip . --exclude "*.pyc" --exclude "__pycache__/*"
          cd ..

      - name: Upload to S3
        run: |
          BUCKET="mpan-validation-codebuild-sandbox-435097195861"
          aws s3 cp code.zip s3://$BUCKET/code.zip
          aws s3 cp docker/tnsnames.ora s3://$BUCKET/tnsnames.ora
          zip source.zip docker/Dockerfile buildspec.yml
          aws s3 cp source.zip s3://$BUCKET/source.zip

      - name: Trigger CodeBuild
        run: |
          BUILD_ID=$(aws codebuild start-build \
            --project-name mpan-validation-build-sandbox \
            --region eu-north-1 \
            --query 'build.id' \
            --output text)

          echo "Build ID: $BUILD_ID"

          while true; do
            STATUS=$(aws codebuild batch-get-builds \
              --ids $BUILD_ID \
              --region eu-north-1 \
              --query 'builds[0].buildStatus' \
              --output text)
            echo "$(date): $STATUS"
            [ "$STATUS" = "SUCCEEDED" ] || [ "$STATUS" = "FAILED" ] || [ "$STATUS" = "STOPPED" ] && break
            sleep 30
          done

          [ "$STATUS" = "SUCCEEDED" ] && echo "✅ Sandbox deployed!" || exit 1
```
